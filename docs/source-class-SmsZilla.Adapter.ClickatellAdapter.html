37      * Send message through Clickatell.com gateway
38      * @param SmsMessageModel $message
39      * @return bool
40      */
41     public function send(MessageInterface $message, $skipErrors = true) {
42 
43         // check clickatell limit
44         if (count($message->getRecipients()) > 600) {
45             $this->addError(new SendingError($message->getRecipients(),
46                     'You have excedded provider limit of messages to send in one time.'
47                     , $message->error));
48         }
49 
50         $gateway = $this->getClient();
51 
52         try {
53             $response = $gateway->sendMessage($message->getRecipients(), $message->getText());
54             foreach ($response as $message) {
55                 if ($message->errorCode) {
56                     $this->addError(new SendingError($message->destination, $message->errorCode, $message->error));
57                 }
58             }
59         }
60         catch (Exception $e) {
61             $this->addError(new SendingError('', $e->getCode(), $e->getMessage()));
62             if (!$skipErrors) {
63                 throw new \RuntimeException($e->getMessage(), $e->getCode());
64             }
65         }
66         return $this->getErrors()->count() === 0;
67     }
68
